### Quick Start

```tsx {1,6} showLineNumbers title="app/page.tsx"
import { SplitTextAnimation } from "@/components/ui/split-text-animation";

export default function Home() {
  return (
    <SplitTextAnimation
      text="Dynamic Scale In Animation"
      animationType="scale"
      duration={0.6}
      stagger={0.04}
    />
  );
}
```

### API Reference

| Prop              | Type      | Default   | Description                   |
| ----------------- | --------- | --------- | ----------------------------- |
| `text`            | `string`  | required  | The text to animate           |
| `animationType`   | `"scale"` | `"scale"` | Type of animation to apply    |
| `duration`        | `number`  | `0.6`     | Animation duration in seconds |
| `stagger`         | `number`  | `0.04`    | Delay between each character  |
| `delay`           | `number`  | `0`       | Initial delay                 |
| `className`       | `string`  | `""`      | Additional CSS classes        |
| `showRerunButton` | `boolean` | `false`   | Show a rerun button           |
