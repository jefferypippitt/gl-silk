## Create a theme provider

```tsx title="components/theme-provider.tsx" showLineNumbers
"use client";

import * as React from "react";
import { ThemeProvider as NextThemesProvider } from "next-themes";

export function ThemeProvider({
  children,
  ...props
}: React.ComponentProps<typeof NextThemesProvider>) {
  return <NextThemesProvider {...props}>{children}</NextThemesProvider>;
}
```

### Wrap your root layout

```tsx title="app/layout.tsx" showLineNumbers
import { ThemeProvider } from "@/components/theme-provider";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <ThemeProvider attribute="class" defaultTheme="system" enableSystem>
          {children}
        </ThemeProvider>
      </body>
    </html>
  );
}
```

### Advanced Example

The ultimate theme toggle combining split text animations, dual colors, and dynamic text!

<Tabs items={['Preview', 'Code']}>

```tsx
import { useTheme } from "next-themes";
import { ThemeToggleButton } from "@/components/ui/theme-toggle-button";
import { SplitTextAnimation } from "@/components/ui/split-text-animation";

export function ThemeToggleAdvancedDemo() {
  const { theme } = useTheme();

  return (
    <ThemeToggleButton
      variant="outline"
      size="lg"
      className="rounded-full [&_svg_circle]:fill-yellow-500 [&_svg_path]:stroke-yellow-500 [&[data-state=checked]_svg_circle]:fill-blue-500 [&[data-state=checked]_svg_path]:stroke-blue-500"
    >
      <SplitTextAnimation
        text={theme === "dark" ? "Light" : "Dark"}
        animationType="fade"
        duration={0.6}
        delay={0.1}
      />
    </ThemeToggleButton>
  );
}
```

### API Reference

| Prop            | Type                                                                          | Default          | Description                                |
| --------------- | ----------------------------------------------------------------------------- | ---------------- | ------------------------------------------ |
| `variant`       | `"default" \| "destructive" \| "outline" \| "secondary" \| "ghost" \| "link"` | `"default"`      | Button variant (same as shadcn Button)     |
| `size`          | `"default" \| "sm" \| "lg" \| "icon" \| "icon-sm" \| "icon-lg"`               | `"default"`      | Button size variant                        |
| `duration`      | `number`                                                                      | `0.5`            | Animation duration in seconds              |
| `ease`          | `string`                                                                      | `"power2.inOut"` | GSAP easing function                       |
| `asChild`       | `boolean`                                                                     | `false`          | Render as child component using Radix Slot |
| `onStateChange` | `(isDark: boolean) => void`                                                   | —                | Callback when state changes                |

The button **automatically integrates with `next-themes`** and toggles between **light** and **dark** modes. When `enableSystem` is set to `true` in the ThemeProvider:

- **System theme is detected automatically** - The button respects the user's OS preference as the default
- **Clicking toggles between light/dark** - Users can override the system preference
- **The button state reflects the current theme** - Shows sun for light, moon for dark
- **Smooth morphing animation** - GSAP morphs between sun rays and moon crescent

The component handles all theme logic internally - just drop it in and it works!
